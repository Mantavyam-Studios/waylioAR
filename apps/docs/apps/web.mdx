# Marketing Website

The marketing website is the public-facing site for Waylio, built with Next.js and optimized for performance and SEO. It serves as the primary marketing channel, providing information about the platform's features, benefits, and pricing.

<Info>
The marketing website runs on port 3001 in development. We recommend deploying it to `www.{yourdomain}.com` in production.
</Info>

## Overview

The marketing website is designed to convert visitors into customers by showcasing Waylio's value proposition, features, and success stories. It's built with modern web technologies and optimized for search engines and performance.

  import { basehub } from '@repo/cms';
  
  export async function generateStaticParams() {
    const { blog } = await basehub({ next: { revalidate: 60 } }).query({
      blog: {
        posts: {
          _slug: true,
        },
      },
    });
    
    return blog.posts.map((post) => ({
      slug: post._slug,
    }));
  }
  
  export default async function BlogPost({ params }) {
    const { blog } = await basehub().query({
      blog: {
        posts: {
          __args: { where: { _slug: { eq: params.slug } } },
          _title: true,
          content: { json: { content: true } },
          publishedAt: true,
          author: { name: true, avatar: { url: true } },
        },
      },
    });
    
    const post = blog.posts[0];
    
    return (
      <article>
        <BlogHeader post={post} />
        <BlogContent content={post.content.json.content} />
      </article>
    );
  }

  export function useLiveContent() {
    const [content, setContent] = useState(null);
    
    useEffect(() => {
      const subscription = basehub.subscribe({
        query: {
          blog: {
            posts: {
              _title: true,
              _slug: true,
              publishedAt: true,
            },
          },
        },
        onUpdate: (data) => setContent(data),
      });
      
      return () => subscription.unsubscribe();
    }, []);
    
    return content;
  }

  import { basehub } from '@repo/cms';
  
  export default async function sitemap() {
    const { blog } = await basehub().query({
      blog: {
        posts: {
          _slug: true,
          publishedAt: true,
        },
      },
    });
    
    const blogPosts = blog.posts.map((post) => ({
      url: `https://waylio.com/blog/${post._slug}`,
      lastModified: post.publishedAt,
      changeFrequency: 'weekly',
      priority: 0.7,
    }));
    
    return [
      {
        url: 'https://waylio.com',
        lastModified: new Date(),
        changeFrequency: 'daily',
        priority: 1,
      },
      {
        url: 'https://waylio.com/features',
        lastModified: new Date(),
        changeFrequency: 'weekly',
        priority: 0.9,
      },
      ...blogPosts,
    ];
  }

## Development

### Local Development

<Steps>
<Step title="Start the Website">
  ```bash
  # Start only the marketing website
  pnpm dev --filter web
  
  # Or start all services
  pnpm dev
  ```
  
  The website will be available at http://localhost:3001
</Step>

<Step title="CMS Setup">
  Configure BaseHub CMS connection:
  
  ```bash
  # Set BaseHub token
  export BASEHUB_TOKEN="your_token_here"
  
  # Test CMS connection
  pnpm cms:sync --filter web
  ```
</Step>

<Step title="Content Preview">
  Enable live content preview:
  
  ```bash
  # Start with draft content
  pnpm dev:preview --filter web
  ```
</Step>
</Steps>
